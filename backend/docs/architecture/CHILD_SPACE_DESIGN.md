# ChildSpace: The Living Portrait

> "×ž×¨×—×‘ ×”×”×‘× ×”" - A space where understanding lives and can be shared

## Philosophy

ChildSpace is not a file drawer. It's a **living portrait** of the child that:
- Shows who the child truly is (not just deficits)
- Documents the journey of discovery
- Empowers parents to share understanding in any language
- Provides actionable, strength-based guidance

def _derive_essence(self, gestalt: LivingGestalt) -> EssenceView:
    """Derive the Essence tab content."""

    # Get narrative from synthesis if available
    narrative = None
    narrative_status = "not_ready"
    if gestalt.synthesis and gestalt.synthesis.essence:
        narrative = gestalt.synthesis.essence.narrative
        narrative_status = "available"

    # Extract strengths from facts
    strengths = []
    strength_facts = [f for f in gestalt.understanding.facts
                     if f.domain == "strengths"]
    for fact in strength_facts:
        strengths.append(StrengthCard(
            domain=infer_strength_domain(fact.content),
            icon=get_strength_icon(fact.content),
            title_he=extract_title(fact.content),
            evidence=fact.content,
            source=fact.source
        ))

    # Add strengths from video observations
    for cycle in gestalt.exploration_cycles:
        for scenario in cycle.video_scenarios:
            if scenario.analysis_result:
                for strength in scenario.analysis_result.strengths_observed:
                    strengths.append(StrengthCard(
                        domain="observed",
                        icon="ðŸŽ¬",
                        title_he=strength,
                        evidence=f"× ×¦×¤×” ×‘×¡×¨×˜×•×Ÿ: {scenario.title}",
                        source="video"
                    ))

    # Active explorations
    active_explorations = []
    for cycle in gestalt.exploration_cycles:
        if cycle.status == "active":
            active_explorations.append(ExplorationSummary(
                id=cycle.id,
                question=cycle.focus,
                theory=cycle.theory,
                confidence=cycle.confidence,
                evidence_count=len(cycle.evidence)

    # Facts by domain

    for fact in gestalt.understanding.facts:
        if fact.domain not in facts_by_domain:
            facts_by_domain[fact.domain] = []
        facts_by_domain[fact.domain].append(fact.content)

    return EssenceView(
        narrative=narrative,
        narrative_status=narrative_status,
        temperament=gestalt.understanding.essence.temperament if gestalt.understanding.essence else [],
        strengths=strengths,
        active_explorations=active_explorations,
        facts_by_domain=facts_by_domain
    )


def _derive_discoveries(self, gestalt: LivingGestalt) -> DiscoveriesView:
    """Derive the Discoveries timeline."""

    milestones = []

    # Add journal entries as milestones
    for entry in gestalt.journal:
        milestones.append(DiscoveryMilestone(
            id=f"journal_{entry.timestamp}",
            timestamp=entry.timestamp,
            type="exploration_began" if "×”×ª×—×œ× ×•" in entry.summary else "insight",
            icon="ðŸ“",
            title_he=entry.summary,
            description_he=", ".join(entry.learned),
            significance="major" if entry.significance == "notable" else "normal"
        ))

    # Add video analyses as milestones
    for cycle in gestalt.exploration_cycles:
        for scenario in cycle.video_scenarios:
            if scenario.status == "analyzed" and scenario.analyzed_at:
                # Find key insight from analysis
                insights = scenario.analysis_result.insights if scenario.analysis_result else []
                description = insights[0] if insights else "× ×™×ª×•×— ×”×¡×¨×˜×•×Ÿ ×”×•×©×œ×"

                milestones.append(DiscoveryMilestone(
                    id=f"video_{scenario.id}",
                    timestamp=scenario.analyzed_at,
                    type="video_analyzed",
                    icon="ðŸŽ¬",
                    title_he=f"×¦×¤×™× ×•: {scenario.title}",
                    description_he=description,
                    video_id=scenario.id,
                    exploration_id=cycle.id,
                    significance="major"
                ))

    # Sort by timestamp, newest first
    milestones.sort(key=lambda m: m.timestamp, reverse=True)

    return DiscoveriesView(
        milestones=milestones,
        days_since_start=calculate_days(gestalt),
        total_conversations=len(gestalt.journal),
        total_videos=count_videos(gestalt),
        insights_discovered=count_insights(gestalt)
    )


def _derive_observations(self, gestalt: LivingGestalt) -> ObservationsView:
    """Derive the Observations (videos) tab."""

    videos = []

    for cycle in gestalt.exploration_cycles:
        for scenario in cycle.video_scenarios:
            if scenario.video_path:  # Has uploaded video

                observations = []
                strengths = []
                insights = []
                effect = None

                if scenario.analysis_result:
                    observations = [
                        TimestampedObservation(
                            content=obs.content,
                            timestamp_start=obs.timestamp_start,
                            timestamp_end=obs.timestamp_end,
                            domain=obs.domain,
                            effect=obs.effect
                        )
                        for obs in scenario.analysis_result.observations
                    ]
                    strengths = scenario.analysis_result.strengths_observed
                    insights = scenario.analysis_result.insights
                    effect = scenario.analysis_result.hypothesis_evidence.overall_verdict \
                             if scenario.analysis_result.hypothesis_evidence else None

                videos.append(VideoCard(
                    id=scenario.id,
                    title=scenario.title,
                    video_path=scenario.video_path,
                    thumbnail_path=None,  # TODO: Generate thumbnails
                    duration_seconds=0,  # TODO: Extract from video
                    uploaded_at=scenario.uploaded_at,
                    status=scenario.status,
                    analyzed_at=scenario.analyzed_at,
                    hypothesis_title=cycle.theory,
                    hypothesis_id=cycle.id,
                    observations=observations,
                    strengths_observed=strengths,
                    insights=insights,
                    evidence_effect=effect
                ))

    # Sort by upload date, newest first
    videos.sort(key=lambda v: v.uploaded_at or datetime.min, reverse=True)

    analyzed = sum(1 for v in videos if v.status == "analyzed")
    pending = sum(1 for v in videos if v.status in ["pending", "uploaded"])

    return ObservationsView(
        videos=videos,
        total_videos=len(videos),
        analyzed_count=analyzed,
        pending_count=pending
    )
